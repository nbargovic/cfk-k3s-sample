# https://github.com/confluentinc/confluent-kubernetes-examples/blob/master/assets/certs/component-certs/README.md

mkdir ./generated && openssl genrsa -out ./generated/rootCAkey.pem 2048

openssl req -x509  -new -nodes \
  -key ./generated/rootCAkey.pem \
  -days 3650 \
  -out ./generated/cacerts.pem \
  -subj "/C=US/ST=CA/L=MVT/O=TestOrg/OU=Cloud/CN=TestCA"

cfssl gencert -ca=./generated/cacerts.pem -ca-key=./generated/rootCAkey.pem -config=./mtls/configs/ca-config.json -profile=server ./mtls/configs/zookeeper-server-domain.json | cfssljson -bare ./generated/zookeeper-server

cfssl gencert -ca=./generated/cacerts.pem -ca-key=./generated/rootCAkey.pem -config=./mtls/configs/ca-config.json -profile=server ./mtls/configs/kafka-server-domain.json | cfssljson -bare ./generated/kafka-server

# https://docs.confluent.io/operator/current/co-security-compliance.html#configure-fips-enabled-ak

cat ./generated/cacerts.pem ./generated/kafka-server.pem > ./generated/kafka-server-chain.pem

keytool -noprompt -keystore ./generated/truststore.bcfks \
    -storetype BCFKS \
    -alias CARoot \
    -import -file ./generated/cacerts.pem \
    -storepass password \
    -keypass password \
    -providerclass org.bouncycastle.jcajce.provider.BouncyCastleFipsProvider \
    -providerpath ./bc-fips/bc-fips-1.0.2.3.jar

openssl pkcs12 -export \
    -in ./generated/kafka-server-chain.pem \
    -inkey ./generated/kafka-server-key.pem \
    -out ./generated/kafka-server.p12 \
    -name kafka-server \
    -passout pass:password

keytool -importkeystore \
    -srckeystore ./generated/kafka-server.p12 \
    -srcstoretype pkcs12 \
    -srcstorepass password \
    -destkeystore ./generated/keystore.bcfks \
    -deststorepass password \
    -destkeypass password \
    -deststoretype BCFKS \
    -providername BCFIPS \
    -providerclass org.bouncycastle.jcajce.provider.BouncyCastleFipsProvider \
    -providerpath ./bc-fips/bc-fips-1.0.2.3.jar

keytool -noprompt -keystore ./generated/keystore.bcfks \
    -storetype BCFKS \
    -alias CARoot \
    -import -file ./generated/cacerts.pem \
    -storepass password \
    -keypass password \
    -providerclass org.bouncycastle.jcajce.provider.BouncyCastleFipsProvider \
    -providerpath ./bc-fips/bc-fips-1.0.2.3.jar

keytool -noprompt -keystore ./generated/truststore.jks \
    -storetype pkcs12 \
    -alias CARoot \
    -import -file ./generated/cacerts.pem \
    -storepass password \
    -keypass password

keytool -importkeystore \
    -srckeystore ./generated/kafka-server.p12 \
    -srcstoretype pkcs12 \
    -srcstorepass password \
    -deststoretype pkcs12 \
    -destkeystore ./generated/keystore.jks \
    -deststorepass password \
    -destkeypass password

keytool -noprompt -keystore ./generated/keystore.jks \
    -alias CARoot \
    -storetype pkcs12 \
    -import -file ./generated/cacerts.pem \
    -storepass password \
    -keypass password

kubectl create secret generic fips-tls \
    --from-file=keystore.bcfks=./generated/keystore.bcfks \
    --from-file=truststore.bcfks=./generated/truststore.bcfks \
    --from-file=keystore.jks=./generated/keystore.jks \
    --from-file=truststore.jks=./generated/truststore.jks \
    --from-literal=jksPassword.txt=jksPassword=password \
    --namespace confluent

kubectl create secret generic tls-zookeeper --from-file=fullchain.pem=./generated/zookeeper-server.pem --from-file=cacerts.pem=./generated/cacerts.pem --from-file=privkey.pem=./generated/zookeeper-server-key.pem --namespace confluent

cfssl gencert -ca=./generated/cacerts.pem -ca-key=./generated/rootCAkey.pem -config=./mtls/configs/ca-config.json -profile=server ./mtls/configs/schemaregistry-server-domain.json | cfssljson -bare ./generated/schemaregistry-server
kubectl create secret generic tls-schemaregistry --from-file=fullchain.pem=./generated/schemaregistry-server.pem --from-file=cacerts.pem=./generated/cacerts.pem --from-file=privkey.pem=./generated/schemaregistry-server-key.pem --namespace confluent

cfssl gencert -ca=./generated/cacerts.pem -ca-key=./generated/rootCAkey.pem -config=./mtls/configs/ca-config.json -profile=server ./mtls/configs/connect-server-domain.json | cfssljson -bare ./generated/connect-server
kubectl create secret generic tls-connect --from-file=fullchain.pem=./generated/connect-server.pem --from-file=cacerts.pem=./generated/cacerts.pem --from-file=privkey.pem=./generated/connect-server-key.pem --namespace confluent
